self.heaven_trigger = {'w': False, 'b': False} def get_piece(self, r, c): if 0 <= r < 8 and 0 <= c < 8: return self.board[r][c] return None def is_valid_move(self, start, end): sr, sc = start er, ec = end piece = self.board[sr][sc] target = self.board[er][ec] if not piece or piece[0] != self.turn: return False if target and target[0] == self.turn: return False dr, dc = er - sr, ec - sc p_type = piece[2:] color = piece[0] if p_type == 'p': direction = -1 if color == 'w' else 1 if "gravity_flip" in self.augments[color]: if abs(dr) == 1 and dc == 0 and not target: return True if dr == direction and dc == 0 and not target: return True if (sr == 6 if color == 'w' else sr == 1) and dr == 2*direction and dc == 0 and not target: if not self.board[sr+direction][sc]: return True if dr == direction and abs(dc) == 1 and target: return True elif p_type == 'k': if "king_attack" in self.augments[color]: if dr == 0 or dc == 0 or abs(dr) == abs(dc): return self.is_path_clear(start, end) return abs(dr) <= 1 and abs(dc) <= 1 elif p_type == 'n': return (abs(dr), abs(dc)) in [(1,2), (2,1)] elif p_type == 'r': if dr == 0 or dc == 0: return self.is_path_clear(start, end) elif p_type == 'b': if abs(dr) == abs(dc): return self.is_path_clear(start, end) elif p_type == 'q': if dr == 0 or dc == 0 or abs(dr) == abs(dc): return self.is_path_clear(start, end) if "double_step" in self.augments[color]: if abs(dr) <= 2 and abs(dc) <= 2: if p_type in ['n', 'k']: return True return self.is_path_clear(start, end) return False def is_path_clear(self, start, end): sr, sc = start er, ec = end dr = 1 if er > sr else (-1 if er < sr else 0) dc = 1 if ec > sc else (-1 if ec < sc else 0) curr_r, curr_c = sr + dr, sc + dc while (curr_r, curr_c) != (er, ec): if self.board[curr_r][curr_c]: return False curr_r += dr curr_c += dc return True def move_piece(self, start, end): sr, sc = start er, ec = end piece = self.board[sr][sc] target = self.board[er][ec] self.history.append({'board': copy.deepcopy(self.board), 'turn': self.turn, 'captured': copy.deepcopy(self.captured), 'move_count': self.move_count}) thorns_kill = target and "thorns" in self.augments[target[0]] if target: self.captured[self.turn].append(target) self.board[er][ec] = piece self.board[sr][sc] = None if thorns_kill: self.board[er][ec] = None game_over = target and target[2:] == 'k' self.turn = 'b' if self.turn == 'w' else 'w' self.move_count += 1 return game_over def undo(self): if not self.history: return prev = self.history.pop() self.board = prev['board'] self.turn = prev['turn'] self.captured = prev['captured'] self.move_count = prev['move_count'] # --- 메인 애플리케이션 --- class GameApp: def __init__(self): self.engine = ChessEngine() self.state = STATE_MENU self.selected = None self.theme_name
