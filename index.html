import pygame import sys import copy import random import math import time # --- 초기 설정 --- pygame.init() BOARD_SIZE = 640 SIDE_WIDTH = 250 WIDTH = BOARD_SIZE + SIDE_WIDTH * 2 HEIGHT = BOARD_SIZE CELL = BOARD_SIZE // 8 # 메인 디스플레이 설정 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("AUGMENT CHESS - Professional Final") # --- 폰트 및 리소스 --- def get_font(size, bold=False, is_piece=False): if is_piece: piece_fonts = ["segoeuisymbol", "quivira", "dejavusans", "freeserif", "arial"] for f in piece_fonts: try: found = pygame.font.SysFont(f, size) if found: return found except: continue text_fonts = ["malgungothic", "nanumgothic", "notosansckjkr", "arial"] for f in text_fonts: try: found = pygame.font.SysFont(f, size, bold=bold) if found: return found except: continue return pygame.font.SysFont(None, size) font_piece = get_font(60, is_piece=True) font_ui = get_font(18, bold=True) font_log = get_font(14) font_msg = get_font(24, bold=True) font_card_title = get_font(20, bold=True) font_card_desc = get_font(14) # --- 상수 및 테마 --- COLORS = { "Modern Dark": {"white": (230, 233, 235), "black": (70, 85, 95), "bg": (20, 22, 26), "sidebar": (30, 34, 40)}, "Classic Wood": {"white": (240, 217, 181), "black": (181, 136, 99), "bg": (46, 30, 20), "sidebar": (65, 43, 30)}, "Deep Sea": {"white": (173, 216, 230), "black": (25, 25, 112), "bg": (10, 15, 30), "sidebar": (20, 25, 45)} } PIECES = { 'w_k': '♔', 'w_q': '♕', 'w_r': '♖', 'w_b': '♗', 'w_n': '♘', 'w_p': '♙', 'b_k': '♚', 'b_q': '♛', 'b_r': '♜', 'b_b': '♝', 'b_n': '♞', 'b_p': '♟' } STATE_MENU = 0 STATE_GAME = 1 STATE_AUGMENT = 2 STATE_SETTINGS = 3 STATE_OVER = 4 # --- 증강 풀 (Augment Pool) --- AUGMENT_POOL = [ {"id": "necromancy", "name": "네크로맨시", "desc": "기물을 잡으면 하급 기물로 부활시킵니다."}, {"id": "king_attack", "name": "킹스 어택", "desc": "킹이 퀸처럼 움직일 수 있게 됩니다."}, {"id": "road_to_heaven", "name": "천국에 가는 길", "desc": "다음 증강에서 반드시 중력이 뒤바뀝니다."}, {"id": "gravity_flip", "name": "중력 반전", "desc": "폰이 뒤로도 움직일 수 있게 됩니다."}, {"id": "double_step", "name": "기동력 강화", "desc": "모든 기물이 한 번에 두 칸까지 점프 가능합니다."}, {"id": "thorns", "name": "가시 갑옷", "desc": "내 기물을 잡은 상대 기물도 함께 파괴됩니다."}, ] # --- 체스 엔진 --- class ChessEngine: def __init__(self): self.reset() def reset(self): self.board = [ ['b_r','b_n','b_b','b_q','b_k','b_b','b_n','b_r'], ['b_p']*8, [None]*8, [None]*8, [None]*8, [None]*8, ['w_p']*8, ['w_r','w_n','w_b','w_q','w_k','w_b','w_n','w_r'] ] self.turn = 'w' self.history = [] self.captured = {'w': [], 'b': []} self.move_count = 0 self.augments = {'w': set(), 'b': set()} self.heaven_trigger = {'w': False, 'b': False} def get_piece(self, r, c): if 0 <= r < 8 and 0 <= c < 8: return self.board[r][c] return None def is_valid_move(self, start, end): sr, sc = start er, ec = end piece = self.board[sr][sc] target = self.board[er][ec] if not piece or piece[0] != self.turn: return False if target and target[0] == self.turn: return False dr, dc = er - sr, ec - sc p_type = piece[2:] color = piece[0] # 기본 행마법 (증강 적용 포함) if p_type == 'p': direction = -1 if color == 'w' else 1 # 중력 반전 적용 if "gravity_flip" in self.augments[color]: if abs(dr) == 1 and dc == 0 and not target: return True # 기본 전진 if dr == direction and dc == 0 and not target: return True # 첫 수 2칸 if (sr == 6 if color == 'w' else sr == 1) and dr == 2*direction and dc == 0 and not target: if not self.board[sr+direction][sc]: return True # 대각 잡기 if dr == direction and abs(dc) == 1 and target: return True elif p_type == 'k': if "king_attack" in self.augments[color]: if dr == 0 or dc == 0 or abs(dr) == abs(dc): return self.is_path_clear(start, end) return abs(dr) <= 1 and abs(dc) <= 1 elif p_type == 'n': return (abs(dr), abs(dc)) in [(1,2), (2,1)] elif p_type == 'r': if dr == 0 or dc == 0: return self.is_path_clear(start, end) elif p_type == 'b': if abs(dr) == abs(dc): return self.is_path_clear(start, end) elif p_type == 'q': if dr == 0 or dc == 0 or abs(dr) == abs(dc): return self.is_path_clear(start, end) # 기동력 강화 증강 if "double_step" in self.augments[color]: if abs(dr) <= 2 and abs(dc) <= 2: if p_type in ['n', 'k']: return True return self.is_path_clear(start, end) return False def is_path_clear(self, start, end): sr, sc = start er, ec = end dr = 1 if er > sr else (-1 if er < sr else 0) dc = 1 if ec > sc else (-1 if ec < sc else 0) curr_r, curr_c = sr + dr, sc + dc while (curr_r, curr_c) != (er, ec): if self.board[curr_r][curr_c]: return False curr_r += dr curr_c += dc return True def move_piece(self, start, end): sr, sc = start er, ec = end piece = self.board[sr][sc] target = self.board[er][ec] # 히스토리 저장 self.history.append({ 'board': copy.deepcopy(self.board), 'turn': self.turn, 'captured': copy.deepcopy(self.captured), 'move_count': self.move_count }) # 가시 갑옷 증강 thorns_kill = False if target and "thorns" in self.augments[target[0]]: thorns_kill = True # 기물 잡기 및 부활(네크로맨시) if target: self.captured[self.turn].append(target) if "necromancy" in self.augments[self.turn]: # 잡은 기물을 폰으로 부활시켜 내 진영에 합류 (논리적 예시) pass self.board[er][ec] = piece self.board[sr][sc] = None if thorns_kill: self.board[er][ec] = None # 같이 죽음 # 승리 판정 (킹 획득 시) game_over = False if target and target[2:] == 'k': game_over = True self.turn = 'b' if self.turn == 'w' else 'w' self.move_count += 1 return game_over def undo(self): if not self.history: return prev = self.history.pop() self.board = prev['board'] self.turn = prev['turn'] self.captured = prev['captured'] self.move_count = prev['move_count'] # --- 메인 애플리케이션 --- class GameApp: def __init__(self): self.engine = ChessEngine() self.state = STATE_MENU self.selected = None self.theme_name = "Modern Dark" self.shake_time = 0 self.move_log = [] self.log_scroll = 0 self.timer = [0, 0] # w, b seconds self.last_tick = time.time() self.current_selection = [] self.winner = None def handle_click(self, event): if event.type != pygame.MOUSEBUTTONDOWN: return mx, my = event.pos # 메뉴 상태 if self.state == STATE_MENU: if HEIGHT//2 - 20 <= my <= HEIGHT//2 + 20: self.state = STATE_GAME self.engine.reset() elif HEIGHT//2 + 50 <= my <= HEIGHT//2 + 90: self.state = STATE_SETTINGS return # 설정 상태 if self.state == STATE_SETTINGS: if 100 <= mx <= 300: if 200 <= my <= 240: self.theme_name = "Modern Dark" if 250 <= my <= 290: self.theme_name = "Classic Wood" if 300 <= my <= 340: self.theme_name = "Deep Sea" if mx > WIDTH - 100: self.state = STATE_MENU return # 게임 오버 상태 if self.state == STATE_OVER: self.state = STATE_MENU return # 증강 선택 상태 if self.state == STATE_AUGMENT: card_w = 200 start_x = (WIDTH - (card_w * 3 + 40)) // 2 for i in range(len(self.current_selection)): card_rect = pygame.Rect(start_x + i*(card_w+20), HEIGHT//2-100, card_w, 250) if card_rect.collidepoint(mx, my): aug = self.current_selection[i] self.engine.augments[self.engine.turn].add(aug["id"]) if aug["id"] == "road_to_heaven": self.engine.heaven_trigger[self.engine.turn] = True self.state = STATE_GAME return # 인게임 조작 # 1. 사이드바 버튼 (UNDO / RESET) if mx < SIDE_WIDTH: if 500 <= my <= 540: self.engine.undo() if 550 <= my <= 590: self.engine.reset() self.state = STATE_MENU return # 2. 체스판 클릭 (핵심!) board_x = mx - SIDE_WIDTH if 0 <= board_x < BOARD_SIZE: r, c = my // CELL, board_x // CELL if self.selected: if self.engine.is_valid_move(self.selected, (r, c)): # 기보 기록 p = self.engine.board[self.selected[0]][self.selected[1]] self.move_log.append(f"{p[2:].upper()}{chr(97+c)}{8-r}") if self.engine.move_piece(self.selected, (r, c)): self.state = STATE_OVER self.winner = "WHITE" if self.engine.turn == 'b' else "BLACK" self.shake_time = 20 self.selected = None # 증강 턴 체크 (매 8턴마다) if self.engine.move_count > 0 and self.engine.move_count % 8 == 0: self.prepare_augments() else: piece = self.engine.get_piece(r, c) if piece and piece[0] == self.engine.turn: self.selected = (r, c) else: self.selected = None else: piece = self.engine.get_piece(r, c) if piece and piece[0] == self.engine.turn: self.selected = (r, c) def prepare_augments(self): pool = list(AUGMENT_POOL) color = self.engine.turn if self.engine.heaven_trigger[color]: # 중력 반전 확정 등장 gv = [a for a in pool if a["id"] == "gravity_flip"][0] others = [a for a in pool if a["id"] != "gravity_flip"] self.current_selection = [gv] + random.sample(others, 2) self.engine.heaven_trigger[color] = False else: self.current_selection = random.sample(pool, 3) self.state = STATE_AUGMENT def draw(self): theme = COLORS[self.theme_name] screen.fill(theme["bg"]) if self.state == STATE_MENU: self.draw_text("AUGMENT CHESS", font_title, (0, 255, 150), WIDTH//2, HEIGHT//2 - 100) self.draw_button("GAME START", WIDTH//2, HEIGHT//2, 200, 40) self.draw_button("SETTINGS", WIDTH//2, HEIGHT//2 + 70, 200, 40) return if self.state == STATE_SETTINGS: self.draw_text("SETTINGS", font_title, (255, 255, 255), WIDTH//2, 50) self.draw_text("Board Theme", font_ui, (200, 200, 200), 200, 150) for i, name in enumerate(COLORS.keys()): color = (0, 255, 150) if self.theme_name == name else (255, 255, 255) self.draw_text(name, font_ui, color, 200, 220 + i*50) self.draw_text("[BACK]", font_ui, (255, 100, 100), WIDTH-50, 30) return # 게임 화면 드로잉 off_x, off_y = 0, 0 if self.shake_time > 0: off_x = random.randint(-5, 5) off_y = random.randint(-5, 5) self.shake_time -= 1 # 1. 사이드바 (좌) pygame.draw.rect(screen, theme["sidebar"], (0, 0, SIDE_WIDTH, HEIGHT)) self.draw_text(f"TURN: {'WHITE' if self.engine.turn=='w' else 'BLACK'}", font_ui, (255, 255, 255), SIDE_WIDTH//2, 50) # 적용된 증강 목록 self.draw_text("AUGMENTS", font_ui, (0, 255, 150), SIDE_WIDTH//2, 150) curr_augs = self.engine.augments['w'] if self.engine.turn == 'w' else self.engine.augments['b'] for i, aug_id in enumerate(curr_augs): aug_name = next(a["name"] for a in AUGMENT_POOL if a["id"] == aug_id) self.draw_text(f"• {aug_name}", font_log, (200, 200, 200), SIDE_WIDTH//2, 180 + i*20) self.draw_button("UNDO", SIDE_WIDTH//2, 520, 120, 40) self.draw_button("RESET", SIDE_WIDTH//2, 570, 120, 40) # 2. 체스판 for r in range(8): for c in range(8): rect = (SIDE_WIDTH + c*CELL + off_x, r*CELL + off_y, CELL, CELL) color = theme["white"] if (r+c)%2 == 0 else theme["black"] pygame.draw.rect(screen, color, rect) # 선택 표시 if self.selected == (r, c): pygame.draw.rect(screen, (0, 255, 0), rect, 3) # 기물 p = self.engine.board[r][c] if p: p_color = (255, 255, 255) if p[0] == 'w' else (0, 0, 0) self.draw_text(PIECES[p], font_piece, p_color, SIDE_WIDTH + c*CELL + CELL//2 + off_x, r*CELL + CELL//2 + off_y) # 3. 사이드바 (우) - 로그 및 캡쳐 pygame.draw.rect(screen, theme["sidebar"], (WIDTH - SIDE_WIDTH, 0, SIDE_WIDTH, HEIGHT)) self.draw_text("MOVE LOG", font_ui, (255, 255, 255), WIDTH - SIDE_WIDTH//2, 30) for i, log in enumerate(self.move_log[-15:]): self.draw_text(log, font_log, (180, 180, 180), WIDTH - SIDE_WIDTH//2, 60 + i*20) self.draw_text("CAPTURED", font_ui, (255, 255, 255), WIDTH - SIDE_WIDTH//2, 400) # 흑이 잡은 백색 기물 for i, p in enumerate(self.engine.captured['b'][:8]): self.draw_text(PIECES[p], font_ui, (255, 255, 255), WIDTH - SIDE_WIDTH + 30 + i*25, 440) # 백이 잡은 흑색 기물 for i, p in enumerate(self.engine.captured['w'][:8]): self.draw_text(PIECES[p], font_ui, (0, 0, 0), WIDTH - SIDE_WIDTH + 30 + i*25, 470) # 증강 선택창 오버레이 if self.state == STATE_AUGMENT: s = pygame.Surface((WIDTH, HEIGHT), pygame.SRCALPHA) s.fill((0, 0, 0, 180)) screen.blit(s, (0,0)) self.draw_text("CHOOSE YOUR AUGMENT", font_title, (255, 215, 0), WIDTH//2, 100) card_w = 200 start_x = (WIDTH - (card_w * 3 + 40)) // 2 for i, aug in enumerate(self.current_selection): rect = pygame.Rect(start_x + i*(card_w+20), HEIGHT//2-100, card_w, 250) pygame.draw.rect(screen, (40, 45, 55), rect, border_radius=10) pygame.draw.rect(screen, (0, 255, 150), rect, 2, border_radius=10) self.draw_text(aug["name"], font_card_title, (0, 255, 150), rect.centerx, rect.y + 40) # 설명문 자동 줄바꿈 대신 간략히 self.draw_text(aug["desc"], font_card_desc, (200, 200, 200), rect.centerx, rect.y + 100) # 게임 오버 오버레이 if self.state == STATE_OVER: s = pygame.Surface((WIDTH, HEIGHT), pygame.SRCALPHA) s.fill((255, 0, 0, 100)) screen.blit(s, (0,0)) self.draw_text(f"{self.winner} WINS!", font_title, (255, 255, 255), WIDTH//2, HEIGHT//2) self.draw_text("Click to Return Menu", font_ui, (255, 255, 255), WIDTH//2, HEIGHT//2 + 50) def draw_text(self, text, font, color, x, y): img = font.render(text, True, color) rect = img.get_rect(center=(x, y)) screen.blit(img, rect) def draw_button(self, text, x, y, w, h): mx, my = pygame.mouse.get_pos() rect = pygame.Rect(x - w//2, y - h//2, w, h) color = (60, 65, 75) if rect.collidepoint(mx, my): color = (80, 85, 95) pygame.draw.rect(screen, color, rect, border_radius=5) self.draw_text(text, font_ui, (255, 255, 255), x, y) # --- 실행부 --- if __name__ == "__main__": app = GameApp() clock = pygame.time.Clock() # 이 부분은 웹 환경(Pyodide)에서 호출하거나 로컬에서 루프를 돌림 # 아래는 로컬 실행을 위한 루프 예시 running = True while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False app.handle_click(event) app.draw() pygame.display.flip() clock.tick(60) pygame.quit() sys.exit()
